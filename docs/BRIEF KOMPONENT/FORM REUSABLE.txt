# BRIEF KOMPONEN: Form Reusable Pattern (Inertia useForm)
# Project: YUBASHOP
# Lokasi: resources/js/Components/Modules/{module}/Form{Module}.tsx
# Tech: @inertiajs/react (useForm) + Tailwind CSS + shadcn/ui + sonner
# Tipe: React component

---

## DESKRIPSI

Pattern form reusable yang digunakan untuk seluruh halaman tambah (POST) dan edit (PUT/PATCH) di panel admin. Menggunakan hook `useForm` bawaan Inertia.js untuk sinkronisasi state form, penanganan loading, serta pemetaan error validation secara otomatis dari Laravel FormRequest.

---

## PATTERN PROPS INTERFACE

```typescript
interface FormProps<T> {
  mode: 'create' | 'edit'
  initialData?: T              // undefined saat create, berisi objek model saat edit
  onSuccess?: () => void       // callback setelah sukses
}
```

---

## ADVANTAGES OF INERTIA `useForm`

1. **Auto Error Binding:** Error validation dari validator Laravel otomatis tersimpan di object `errors` tanpa perlu manual parsing.
2. **Built-in States:** Memiliki state `processing` (loading indicator), `wasSuccessful`, dan helper `setData`.
3. **SPA Navigation:** Otomatis melakukan partial reload atau redirection pasca-submit tanpa page reload.

## CONTOH IMPLEMENTASI

```tsx
// resources/js/Components/Modules/Games/FormGame.tsx
import { useForm } from '@inertiajs/react'
import { toast } from 'sonner'
import { FormProps } from '@/Types'
import { Button } from '@/Components/UI/button'
import { Input } from '@/Components/UI/input'

interface GameFormData {
  name: string
  sub_name: string
  slug: string
  is_popular: boolean
}

export function FormGame({ mode, initialData, onSuccess }: FormProps<Game>) {
  const { data, setData, post, put, processing, errors, reset } = useForm<GameFormData>({
    name: initialData?.name || '',
    sub_name: initialData?.sub_name || '',
    slug: initialData?.slug || '',
    is_popular: initialData?.is_popular || false,
  })

  function onSubmit(e: React.FormEvent) {
    e.preventDefault()
    
    const options = {
      onStart: () => toast.loading('Menyimpan data...', { id: 'form-save' }),
      onSuccess: () => {
        toast.success(mode === 'create' ? 'Game berhasil ditambahkan' : 'Game berhasil diperbarui', { id: 'form-save' })
        onSuccess?.()
      },
      onError: (errs) => {
        toast.error('Gagal menyimpan data. Periksa inputan Anda.', { id: 'form-save' })
      }
    }

    if (mode === 'create') {
      post(route('admin.games.store'), options)
    } else {
      // Menggunakan PUT/PATCH route web Laravel
      put(route('admin.games.update', initialData!.id), options)
    }
  }

  return (
    <form onSubmit={onSubmit} className="space-y-6">
      <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
        {/* Name Input */}
        <div>
          <label className="text-sm font-medium text-slate-300">Nama Game <span className="text-red-500">*</span></label>
          <Input 
            value={data.name} 
            onChange={e => setData('name', e.target.value)} 
            className={errors.name ? 'border-red-500' : ''}
          />
          {errors.name && <span className="text-red-400 text-xs">{errors.name}</span>}
        </div>
        
        {/* Sub Name Input */}
        <div>
          <label className="text-sm font-medium text-slate-300">Developer <span className="text-red-500">*</span></label>
          <Input 
            value={data.sub_name} 
            onChange={e => setData('sub_name', e.target.value)} 
            className={errors.sub_name ? 'border-red-500' : ''}
          />
          {errors.sub_name && <span className="text-red-400 text-xs">{errors.sub_name}</span>}
        </div>
      </div>
      
      <div className="flex justify-end gap-3">
        <Button type="submit" disabled={processing} className="bg-emerald-600 hover:bg-emerald-700">
          {processing ? 'Menyimpan...' : 'Simpan Data'}
        </Button>
      </div>
    </form>
  )
}
```

## CATATAN PENTING

- **Upload File:** Jika ada input file (gambar/banner game), gunakan native HTML `<input type="file">` dan set data file di `setData`. Inertia otomatis mengonversi request menjadi `Multipart FormData` saat mengirim data file.
- **CSRF Token:** Token CSRF dikelola secara otomatis oleh header Axios/Inertia, tidak perlu disuntikkan manual pada payload form.
